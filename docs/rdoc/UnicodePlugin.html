def UnicodePlugin.reconfigure_filter(bot)
  debug "configuring encodings"
  charset = bot.config['encoding.charset']
  if bot.config['encoding.enable']
    bot.socket.filter = UnicodeFilter.new charset
  else
    bot.socket.filter = nil
  end
end
