def Regexp.new_list(reg, pfx = "")
  if pfx.kind_of?(String) and pfx.empty?
    return %r(#{reg}(?:,?(?:\s+and)?\s+#{reg})*)
  else
    return %r(#{reg}(?:,?(?:\s+and)?(?:\s+#{pfx})?\s+#{reg})*)
  end
end

def mm_cleanup
  new = self.source.gsub(/(^|[^\\])((?:\\\\)*)\(([^?])/) {
    "%s%s(?:%s" % [$1, $2, $3]
  }.sub(/^\^/,'').sub(/\$$/,'')
  Regexp.new(new, self.options)
end
