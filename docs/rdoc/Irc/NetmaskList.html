def [](*args)
  if args.length == 1
    case args[0]
    when Netmask
      self.find { |mask|
        mask.matches?(args[0])
      }
    when String
      self.find { |mask|
        mask.matches?(args[0].to_irc_netmask(:casemap => mask.casemap))
      }
    when Regexp
      self.find { |mask|
        mask.fullform =~ args[0]
      }
    else
      super(*args)
    end
  else
    super(*args)
  end
end
