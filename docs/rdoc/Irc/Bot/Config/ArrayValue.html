def add(val)
  newval = self.get.dup
  unless newval.include? val
    newval << val
    validate_item(val) or raise ArgumentError, "invalid item: #{val}"
    validate(newval) or raise ArgumentError, "invalid value: #{newval.inspect}"
    set(newval)
  end
end

def rm(val)

  set(curval - [val])
end
