def Irc.color(fg=nil,bg=nil)

  return str

def Irc.find_color(data)
  "%02d" % if Integer === data
    data
  else
    f = if String === data
          data.intern
        else
          data
        end
    if ColorCode.key?(f)
      ColorCode[f]
    else
      0
    end
  end
end
